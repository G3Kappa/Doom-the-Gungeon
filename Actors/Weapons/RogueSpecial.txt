#include "Actors/Weapons/Projectiles/GreenLaser.txt"

class RogueSpecialAmmo : Ammo 
{
	Default
	{
		Inventory.MaxAmount 8;
	}
}

Class RogueSpecial : WeaponBase replaces Chainsaw
{
	Default 
	{
		Inventory.PickupMessage "Underhanded and efficient";
		Weapon.SlotNumber 2;
		Weapon.AmmoGive 8;
		Weapon.AmmoUse 1;
		Weapon.AmmoType "RogueSpecialAmmo";
	}
	States
	{
		Spawn:
			RSGN A -1;
			Stop;
		Ready:
			RSGN A 1 A_WeaponReady(WRF_ALLOWRELOAD);
			Loop;
		Select:
			RSGN A 1 A_Raise;
			Loop;
			
		Deselect:
			RSGN A 1 A_Lower;
			Loop;
		
		Reload:
			RSGN R 0 A_JumpIf(CountInv("RogueSpecialAmmo") == 8, "Ready");
			RSGN R 42 Offset(0, 48) A_PlaySound("RogueSpecial/Reload", 1);
			RSGN R 0 A_GiveInventory("RogueSpecialAmmo", 8);
			Goto Ready;
		Fire:
			RSGN A 0 A_GunFlash;
			RSGN A 0 A_PlaySound("RogueSpecial/Fire", 1);
			RSGN A 0 A_FireProjectile("GreenLaser", frandom(-6, 6), 1, 14, 4, 0, frandom(-1.3, 1.3));
			RSGN B 1 Offset(5,40);
			RSGN B 1 Offset(15,48);
			RSGN B 2 Offset(20,44);
			RSGN C 1 Offset(18,40);
			RSGN C 1 Offset(12,37);
			RSGN D 1 Offset(8,35);
			RSGN D 1 Offset(4,32);
			RSGN D 1 Offset(0,32);
			RSGN A 0 A_JumpIf(CountInv("RogueSpecialAmmo") == 0, "Reload");
			Goto Ready;
			
		Flash:
			RSGN A 2 BRIGHT Light("GREENLASER0");
			RSGN B 2 BRIGHT Light("GREENLASER1");
			TNT1 A 1 Light("GREENLASER0");
			Goto LightDone;
	}
}
