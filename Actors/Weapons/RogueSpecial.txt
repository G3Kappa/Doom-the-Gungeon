#include "Actors/Weapons/Projectiles/GreenLaser.txt"

class RogueSpecialAmmo : Ammo 
{
	Default
	{
		Inventory.MaxAmount 8;
	}
}

Class RogueSpecial : WeaponBase replaces Chainsaw
{
	Default 
	{
		Inventory.PickupMessage "Underhanded and efficient";
		Weapon.SlotNumber 2;
		Weapon.AmmoGive 8;
		Weapon.AmmoUse 1;
		Weapon.AmmoType "RogueSpecialAmmo";
	}
	States
	{
		Spawn:
			SHOT A -1;
			Stop;
		Ready:
			SHTG B 1 A_WeaponReady(WRF_ALLOWRELOAD);
			Loop;
		Select:
			SHTG A 1 A_Raise;
			Loop;
			
		Deselect:
			SHTG A 1 A_Lower;
			Loop;
		
		Reload:
			SHTG A 0 A_JumpIf(CountInv("RogueSpecialAmmo") == 8, "Ready");
			SHTG B 60;
			SHTG A 0 A_GiveInventory("RogueSpecialAmmo", 8);
			Goto Ready;
		Fire:
			SHTG A 0 A_GunFlash;
			SHTG A 0 A_PlaySound("RogueSpecial/Fire");
			SHTG A 1 A_FireProjectile("GreenLaser", frandom(-6, 6), 1, 0.5, 0, 0, frandom(-1.3, 1.3));
			SHTG A 1 Offset(0,40);
			SHTG A 1 Offset(0,48);
			SHTG A 1 Offset(0,44);
			SHTG A 1 Offset(0,40);
			SHTG A 1 Offset(0,37);
			SHTG A 1 Offset(0,35);
			SHTG A 6;
			SHTG A 0 A_JumpIf(CountInv("RogueSpecialAmmo") == 0, "Reload");
			Goto Ready;
			
		Flash:
			SHTF A 2 BRIGHT Light("GREENLASER0");
			SHTF B 2 BRIGHT Light("GREENLASER1");
			TNT1 A 1 Light("GREENLASER0");
			Goto LightDone;
	}
}
