Class ElephantBullets : ProjectileBase 
{
	override void PostBeginPlay()
	{
		Super.PostBeginPlay();
		ScatterCount += 5;
		ScatterDamage = 1;
	}

	Default 
	{
		Radius 2;
		Height 16;
		Speed 25;
		ProjectileBase.BaseDamage 5.5;
		Species "PlayerProjectile";
		DamageType "PlayerDamage";
		Decal "BulletChip";
		BounceSound "RogueSpecial/Impact";
		+THRUSPECIES;
	}
	
	States
	{
		Spawn:
			TNT1 A 2;
			TNT1 A 0 A_PostSpawn;
			ELBL AAAAAAAAAAAAAAAAAAAAAA 1 Bright;
			Goto Crash;
			
		XDeath:
		Death:
		Crash:
			ELIM A 0 {
				A_PlaySound("RogueSpecial/Impact", 0);
				Vel = (0, 0, 0); // Stop!
			}
			ELIM A 1 A_PreDeath;
			ELIM A 2 Bright A_SetScale(0.6);
			ELIM A 2 Bright A_SetScale(0.7);
			ELIM AB 2 Bright;
			ELIM B 1 Bright;
			ELIM B 1 Bright;
			ELIM B 1 Bright;
			Stop;
	}
}
