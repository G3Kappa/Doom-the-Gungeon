Class Rebar : ProjectileBase 
{
	override void PostBeginPlay()
	{
		Super.PostBeginPlay();
		BounceCount += 40;
	}
	
	Default 
	{
		Radius 2;
		Height 16;
		Speed 100;
		DamageFunction (50);
		Projectile;
		Species "PlayerProjectile";
		DamageType "PlayerDamage";
		Decal "BulletChip";
		BounceSound "RogueSpecial/Impact";
		+THRUSPECIES;
		+ROCKETTRAIL;
		+RIPPER;
	}
	
	States
	{
		Spawn:
			RSGL A 70 Bright Light("L_GreenLaser");
			Goto Crash;
			
		Death:
		Crash:
			RSIM A 0 A_PlaySound("RogueSpecial/Impact", 0);
			RSIM A 2 Bright A_SetScale(0.6);
			RSIM B 2 Bright;
			RSIM BC 1 Bright;
			RSIM C 1 Bright A_SetScale(0.5);
			RSIM C 1 Bright A_SetScale(0.4);
			RSIM C 1 Bright A_SetScale(0.2);
			Stop;
			
		XDeath:
			TNT1 A 0;
			Stop;
	}
}
